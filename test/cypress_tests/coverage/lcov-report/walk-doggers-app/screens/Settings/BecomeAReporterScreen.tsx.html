    const confirm = () => {
        const jwt = getJwt();

            method: "POST",
            headers: {
                "accept": "application/json",
                'Authorization': 'Bearer ' + jwt
            },
        }).then(async response => {
            if (response.ok) {
                getRoles().then(() => {
                    alert("You are now a reporter");

            }
